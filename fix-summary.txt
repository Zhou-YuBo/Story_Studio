Found the exact issue in both WorldView.vue and InspirationView.vue! Both files use a static HTML canvas overlay for drawing but don't apply the VueFlow viewport transform.\n\nThe problem:\n1. Drawings are stored in flow coordinates (relative to the unpanned canvas)\n2. Rendered directly to the canvas without accounting for zoom/pan\n3. When users zoom/pan, drawings stay fixed in place\n\nThe fix requires adding viewport transform application in the renderDrawings function:\n\n// Add to useVueFlow destructuring:\nconst { ..., viewport } = useVueFlow()\n\nfunction renderDrawings() {\n  if (!canvasCtx.value || !drawingCanvasRef.value) return\n  const { x: vx, y: vy, zoom: vz } = viewport.value\n  \n  canvasCtx.value.clearRect(0, 0, drawingCanvasRef.value.width, drawingCanvasRef.value.height)\n  \n  // Apply viewport transform\n  canvasCtx.value.save()\n  canvasCtx.value.translate(vx, vy)\n  canvasCtx.value.scale(vz, vz)\n  \n  // Draw all shapes\n  for (const draw of store.drawings) { renderDrawingItem(draw) }\n  if (store.drawingPreview) { renderDrawingItem(store.drawingPreview) }\n  \n  canvasCtx.value.restore()\n}
